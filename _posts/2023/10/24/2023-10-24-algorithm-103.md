## 풀이
```javascript
function solution(s) {
    return s
        .slice(2, s.length-2)
        .split("},{")
        .sort((a, b) => a.length - b.length)
        .reduce((acc, cur) => {
            let arr = cur.split(',');
            for (let n of arr) {
                if (!acc.includes(n)) {
                    acc.push(n);
                    return acc;
                } 
            }
        }, [])
        .map(v => parseInt(v));
}
```
